\begin{equation*} \cdots \longrightarrow C_{n+1} \stackrel{d_{n+1}}{\longrightarrow} C_{n} \stackrel{d_{n}}{\longrightarrow} C_{n-1} \longrightarrow \cdots \end{equation*}
