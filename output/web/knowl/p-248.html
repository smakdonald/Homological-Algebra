\begin{equation*} h_{n}(a)=h_{n} \partial_{n+1}^{F}(b)=\partial_{n+1}^{G} h_{n+1}(b) \in \operatorname{im} \partial_{n+1}^{G}=B_{n}\left(G_{\bullet}\right) \end{equation*}
